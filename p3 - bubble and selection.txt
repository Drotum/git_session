import time
import random
import matplotlib.pyplot as plt

# Bubble Sort
def bubble_sort(arr):
    n = len(arr)
    start_time = time.time()
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    end_time = time.time()
    return end_time - start_time

# Selection Sort
def selection_sort(arr):
    n = len(arr)
    start_time = time.time()
    for i in range(n):
        min_idx = i
        for j in range(i+1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    end_time = time.time()
    return end_time - start_time

# Test and plot
n_values = [10, 100, 1000]
bubble_times = []
selection_times = []

for n in n_values:
    arr = random.sample(range(n*2), n)
    bubble_times.append(bubble_sort(arr.copy()))
    selection_times.append(selection_sort(arr.copy()))

plt.plot(n_values, bubble_times, label='Bubble Sort')
plt.plot(n_values, selection_times, label='Selection Sort')
plt.xlabel('Number of elements (n)')
plt.ylabel('Time taken (seconds)')
plt.legend()
plt.show()
