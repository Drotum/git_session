Step-by-Step Instructions
Create a new Java Project in Eclipse:

Open Eclipse IDE.
Go to File > New > Java Project.
Enter the project name (e.g., MissedCallsSimulator).
Create the IncomingCall Class:

Right-click on src in your project and select New > Class.
Name it IncomingCall and paste the following code:
java
Copy code
import java.time.LocalDateTime;

public class IncomingCall {
    private String phoneNumber;
    private String callerName;
    private LocalDateTime timeOfCall;

    public IncomingCall(String phoneNumber, String callerName) {
        this.phoneNumber = phoneNumber;
        this.callerName = callerName != null ? callerName : "private caller";
        this.timeOfCall = LocalDateTime.now();
    }

    public String getPhoneNumber() {
        return phoneNumber;
    }

    public String getCallerName() {
        return callerName;
    }

    public LocalDateTime getTimeOfCall() {
        return timeOfCall;
    }

    @Override
    public String toString() {
        return "IncomingCall{" +
                "phoneNumber='" + phoneNumber + '\'' +
                ", callerName='" + callerName + '\'' +
                ", timeOfCall=" + timeOfCall +
                '}';
    }
}


Create the MissedCallsManager Class:
Right-click on src in your project and select New > Class.
Name it MissedCallsManager and paste the following code:

Copy code
import java.util.LinkedList;

public class MissedCallsManager {
    private LinkedList<IncomingCall> missedCalls;
    private final int MAX_CALLS = 10;

    public MissedCallsManager() {
        missedCalls = new LinkedList<>();
    }

    public void addMissedCall(String phoneNumber, String callerName) {
        if (missedCalls.size() == MAX_CALLS) {
            missedCalls.removeFirst();
        }
        missedCalls.add(new IncomingCall(phoneNumber, callerName));
    }

    public void displayMissedCalls() {
        for (int i = 0; i < missedCalls.size(); i++) {
            System.out.println((i + 1) + ". " + missedCalls.get(i).getPhoneNumber());
        }
    }

    public void displayCallDetails(int index) {
        if (index >= 0 && index < missedCalls.size()) {
            System.out.println(missedCalls.get(index));
        } else {
            System.out.println("Invalid call index.");
        }
    }

    public void deleteCall(int index) {
        if (index >= 0 && index < missedCalls.size()) {
            missedCalls.remove(index);
            System.out.println("Call deleted.");
        } else {
            System.out.println("Invalid call index.");
        }
    }

    public void displayAllCalls() {
        for (IncomingCall call : missedCalls) {
            System.out.println(call);
        }
    }
}

import java.util.Scanner;

public class MissedCallsTester {
    public static void main(String[] args) {
        MissedCallsManager manager = new MissedCallsManager();
        Scanner scanner = new Scanner(System.in);
        boolean running = true;

        // Simulate adding some missed calls
        manager.addMissedCall("1234567890", "John Doe");
        manager.addMissedCall("0987654321", null);
        manager.addMissedCall("5555555555", "Jane Smith");

        while (running) {
            System.out.println("\nMissed Calls:");
            manager.displayMissedCalls();
            System.out.println("Select an option:");
            System.out.println("1. Display call details");
            System.out.println("2. Delete call");
            System.out.println("3. Display all calls");
            System.out.println("4. Exit");

            int choice = scanner.nextInt();
            switch (choice) {
                case 1:
                    System.out.print("Enter call number to display details: ");
                    int displayIndex = scanner.nextInt() - 1;
                    manager.displayCallDetails(displayIndex);
                    break;
                case 2:
                    System.out.print("Enter call number to delete: ");
                    int deleteIndex = scanner.nextInt() - 1;
                    manager.deleteCall(deleteIndex);
                    break;
                case 3:
                    System.out.println("All missed calls:");
                    manager.displayAllCalls();
                    break;
                case 4:
                    running = false;
                    break;
                default:
                    System.out.println("Invalid choice.");
            }
        }
        scanner.close();
    }
}


Run the Program:
Right-click on MissedCallsTester and select Run As > Java Application.
