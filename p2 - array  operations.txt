import matplotlib.pyplot as plt
import time

# a. Sum of all elements in an array
def sum_array(arr):
    total = 0
    for num in arr:
        total += num
    return total

# b. Binary equivalent of a decimal number
def decimal_to_binary(n):
    if n == 0:
        return "0"
    binary = ""
    while n > 0:
        remainder = n % 2
        binary = str(remainder) + binary
        n = n // 2
    return binary

# c. Print even elements of a matrix
def print_even_elements(matrix):
    even_elements = []
    for row in matrix:
        for element in row:
            if element % 2 == 0:
                even_elements.append(element)
    return even_elements

# Test and plot sum of array
n_values = [10, 100, 1000, 10000]
times = []

for n in n_values:
    arr = list(range(n))
    start_time = time.time()
    sum_array(arr)
    end_time = time.time()
    times.append(end_time - start_time)

plt.plot(n_values, times, label='Sum of Array Elements')
plt.xlabel('Number of elements (n)')
plt.ylabel('Time taken (seconds)')
plt.legend()
plt.show()
